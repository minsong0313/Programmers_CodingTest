SELECT II.ITEM_ID, II.ITEM_NAME, II.RARITY
FROM ITEM_INFO II JOIN ITEM_TREE IT ON II.ITEM_ID = IT.ITEM_ID
WHERE IT.PARENT_ITEM_ID IN -- 0,0,1,1(PARENT_ID) -> 1,2,3,4(ITEM_ID)
    (SELECT ITEM_ID -- 0,1,3,4 => 결과적으로 3,4도 RARE이지만 쓰이지 않음
     FROM ITEM_INFO
     WHERE RARITY = 'RARE'
    )
ORDER BY II.ITEM_ID DESC

-- WHERE절 서브쿼리에서 업그레이드 전의 RARITY가 RARE인 아이템을 찾아줘야함
